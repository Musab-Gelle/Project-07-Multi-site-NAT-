Project 7 Config:


HQ Router Project 7 NAT Lab


enable
configure terminal

! VLAN / LAN interfaces
interface g0/0
 ip address 192.168.10.1 255.255.255.0
 ip nat inside
 no shutdown
exit

interface g0/0.20
 ip address 192.168.20.1 255.255.255.0
 ip nat inside
exit

interface g0/0.30
 ip address 192.168.30.1 255.255.255.0
 ip nat inside
exit

! WAN interface to Branch
interface g0/1
 ip address 10.0.0.1 255.255.255.0
 ip nat inside
 no shutdown
exit

! WAN interface to ISP
interface g0/2
 ip address 203.0.113.2 255.255.255.252
 ip nat outside
 no shutdown
exit

! NAT ACL for all internal networks
access-list 1 permit 192.168.10.0 0.0.0.255
access-list 1 permit 192.168.20.0 0.0.0.255
access-list 1 permit 192.168.30.0 0.0.0.255
access-list 1 permit 192.168.40.0 0.0.0.255

! NAT overload (PAT)
ip nat inside source list 1 interface g0/2 overload

! Block inbound traffic from Internet
access-list 199 permit icmp any any echo-reply
access-list 199 permit tcp any any established
access-list 199 deny ip any 192.168.0.0 0.0.255.255
access-list 199 permit ip any any

interface g0/2
 ip access-group 199 in

! Default route to ISP
ip route 0.0.0.0 0.0.0.0 203.0.113.1

end
write memory
